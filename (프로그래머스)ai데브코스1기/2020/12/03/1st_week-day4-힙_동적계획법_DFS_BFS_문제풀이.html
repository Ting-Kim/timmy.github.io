파이썬에서 힙(라이브러리)을 사용하는 방법

import heapq
heapq.heapify(L)     # L(리스트)로부터 min heap을 구성한다.
m = heapq.heappop(L) # min heap인 L에서 최소값을 삭제하고 반환한다.
heapq.heappush(L,x)  # min heap인 L에 원소 x를 삽입한다.
